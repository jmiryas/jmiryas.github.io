# Mengubah Cart Provider

---

Klik `cart_provider.dart`, lalu sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:flutter/material.dart';

import '../models/cart_model.dart';
import '../models/product_model.dart';

class CartProvider extends ChangeNotifier {
  // ignore: prefer_final_fields
  Map<String, CartModel> _carts = {};

  // Mendapatkan produk di cart

  Map<String, CartModel> getCarts() {
    return _carts;
  }

  // Mendapatkan banyak dari produk di cart

  int getCartLength() {
    return _carts.length;
  }

  // Mendapatkan total akhir dari produk di cart

  double getTotalAmount() {
    double total = 0.0;

    _carts.forEach((key, value) {
      total += value.quantity * value.price;
    });

    return total;
  }

  // Menambahkan produk ke cart

  void addProductToCart(ProductModel product) {
    if (_carts.containsKey(product.id)) {
      _carts.update(
        product.id,
        (value) => CartModel(
          id: value.id,
          name: value.name,
          price: value.price,
          quantity: value.quantity + 1,
        ),
      );
    } else {
      _carts.putIfAbsent(
        product.id,
        () => CartModel(
          id: DateTime.now().toString(),
          name: product.name,
          price: product.price,
          quantity: 1,
        ),
      );
    }

    notifyListeners();
  }

  // Remove produk dari cart berdasarkan id

  void removeById(String id) {
    _carts.removeWhere((key, value) => value.id == id);

    notifyListeners();
  }

  void decreaseCartItemById(String id) {
    if (!_carts.containsKey(id)) {
      return;
    }

    if (_carts[id]!.quantity > 1) {
      _carts.update(
        id,
        (value) => CartModel(
            id: value.id,
            name: value.name,
            price: value.price,
            quantity: value.quantity - 1),
      );
    } else {
      _carts.remove(id);
    }

    notifyListeners();
  }

  void clear() {
    _carts.clear();

    notifyListeners();
  }
}
```

Perhatikan pada `decreaseCartItemById`, di sini kita membuat pengecekan apakah sebuah produk dengan id "X" sudah ada di dalam list atau belum. Jika belum maka tidak perlu melakukan apapun.

Jika ada, maka lakukan pengecekan kedua, yaitu apakah quantity-nya lebih dari satu atau tidak. Jika iya, maka kita hanya perlu mengurangi quantity-nya, jika hanya ada satu produk, maka hapus produk tersebut dari dalam list.

# Mengubah Product Widget

---

Klik `product_widget.dart`, lalu sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';
import 'package:cached_network_image/cached_network_image.dart';

import '../models/product_model.dart';
import '../providers/cart_provider.dart';
import '../providers/product_provider.dart';
import '../screens/product_detail_screen.dart';

class ProductWidget extends StatelessWidget {
  final ProductModel product;

  const ProductWidget({
    Key? key,
    required this.product,
  }) : super(key: key);

  @override
  Widget build(BuildContext context) {
    final cartProvider = Provider.of<CartProvider>(context);

    return ClipRRect(
      borderRadius: BorderRadius.circular(10.0),
      child: GridTile(
        footer: GridTileBar(
          backgroundColor: Colors.black54,
          leading: Consumer<ProductProvider>(builder: (context, value, child) {
            return IconButton(
              onPressed: () {
                context.read<ProductProvider>().toggleFavorite(product.id);

                if (value.getProductStatus(product.id)) {
                  ScaffoldMessenger.of(context).showSnackBar(
                    const SnackBar(
                      content: Text("Produk ditambahkan ke favorit"),
                    ),
                  );
                } else {
                  ScaffoldMessenger.of(context).showSnackBar(
                    const SnackBar(
                      content: Text("Produk dihapus dari favorit"),
                    ),
                  );
                }
              },
              icon: Icon(
                product.isFavorite ? Icons.favorite : Icons.favorite_border,
                color: Colors.blue.shade300,
              ),
            );
          }),
          title: Text(product.name),
          trailing: IconButton(
            onPressed: () {
              cartProvider.addProductToCart(product);

              ScaffoldMessenger.of(context).hideCurrentSnackBar();

              ScaffoldMessenger.of(context).showSnackBar(
                SnackBar(
                  content: const Text("Produk ditambahkan ke cart"),
                  duration: const Duration(seconds: 2),
                  action: SnackBarAction(
                      label: "BATALKAN",
                      onPressed: () {
                        cartProvider.decreaseCartItemById(product.id);
                      }),
                ),
              );
            },
            icon: Icon(
              Icons.shopping_cart,
              color: Colors.blue.shade300,
            ),
          ),
        ),
        child: GestureDetector(
          child: CachedNetworkImage(
            imageUrl: product.imageUrl,
            placeholder: (context, url) => const SizedBox(
              width: 30.0,
              height: 30.0,
              child: Center(
                child: CircularProgressIndicator(),
              ),
            ),
            fit: BoxFit.cover,
          ),
          onTap: () {
            Navigator.pushNamed(
              context,
              ProductDetailScreen.routeName,
              arguments: product.id,
            );
          },
        ),
      ),
    );
  }
}
```

Perhatikan, pada saat kita menambahkan produk ke cart, maka kita tambahkan `SnackBarAction` dengan label "BATALKAN". Di sini, kita hanya perlu memanggil fungsi yang sudah kita buat tadi.
