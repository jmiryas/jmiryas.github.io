# Mengubah Cart Provider

---

Buka `cart_provider.dart`, lalu sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:flutter/material.dart';

import '../models/cart_model.dart';
import '../models/product_model.dart';

class CartProvider extends ChangeNotifier {
  Map<String, CartModel> _carts = {};

  Map<String, CartModel> getCarts() {
    return _carts;
  }

  int getCartLength() {
    return _carts.length;
  }

  double getTotalAmount() {
    double total = 0.0;

    _carts.forEach((key, value) {
      total += value.quantity * value.price;
    });

    return total;
  }

  void addProductToCart(ProductModel product) {
    if (_carts.containsKey(product.id)) {
      _carts.update(
        product.id,
        (value) => CartModel(
          id: value.id,
          name: value.name,
          price: value.price,
          quantity: value.quantity + 1,
        ),
      );
    } else {
      _carts.putIfAbsent(
        product.id,
        () => CartModel(
          id: DateTime.now().toString(),
          name: product.name,
          price: product.price,
          quantity: 1,
        ),
      );
    }

    notifyListeners();
  }
}
```

Perhatikan pada fungsi `getTotalAmount`. Fungsi ini berguna untuk mengembalikan total dari seluruh produk yang ada di dalam cart.

# Mengubah Main

---

Sekarang, buka `main.dart`, lalu tambahkan route untuk cart screen, seperti di bawah ini:

```dart
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';

import '../screens/cart_screen.dart';
import '../screens/product_screen.dart';
import '../providers/cart_provider.dart';
import '../providers/product_provider.dart';
import '../screens/product_detail_screen.dart';

void main() {
  runApp(
    MultiProvider(
      providers: [
        ChangeNotifierProvider(
          create: ((context) => ProductProvider()),
        ),
        ChangeNotifierProvider(
          create: ((context) => CartProvider()),
        )
      ],
      child: const MyApp(),
    ),
  );
}

class MyApp extends StatelessWidget {
  const MyApp({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: "Shop App",
      home: const ProductScreen(),
      routes: {
        ProductDetailScreen.routeName: (context) => const ProductDetailScreen(),
        CartScreen.routeName: (context) => const CartScreen(),
      },
    );
  }
}
```

# Membuat Cart Screen

---

Pada folder `screens`, buat sebuah file baru dengan nama `cart_screen.dart`, sesuaikan source code-nya seperti di bawah ini:

```dart
import 'package:intl/intl.dart';
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';

import '../providers/cart_provider.dart';

class CartScreen extends StatelessWidget {
  static String routeName = "/cart";

  const CartScreen({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    double height = MediaQuery.of(context).size.height - 40.0 - 55.0;

    final numberFormat = NumberFormat("#,##0.00", "id_ID");

    return Scaffold(
      appBar: AppBar(
        toolbarHeight: 55.0,
        centerTitle: true,
        title: const Text("Cart"),
      ),
      body: Consumer<CartProvider>(builder: ((context, value, child) {
        return value.getCartLength() > 0
            ? Padding(
                padding: const EdgeInsets.all(20.0),
                child: Column(
                  children: [
                    SizedBox(
                      height: 0.5 * height,
                      child: ListView(
                        children: value.getCarts().entries.map((item) {
                          return Column(
                            children: [
                              ListTile(
                                title: Text(
                                  item.value.name,
                                  style: const TextStyle(
                                    fontWeight: FontWeight.bold,
                                  ),
                                ),
                                subtitle: Text(
                                    "Rp ${numberFormat.format(item.value.price)}"),
                                trailing: Text("${item.value.quantity}"),
                              ),
                              const Divider(
                                color: Colors.black26,
                              )
                            ],
                          );
                        }).toList(),
                      ),
                    ),
                    const SizedBox(
                      height: 20.0,
                    ),
                    Container(
                      height: 0.1 * height,
                      color: Colors.grey.shade300,
                      padding: const EdgeInsets.symmetric(horizontal: 10.0),
                      child: Row(
                        mainAxisAlignment: MainAxisAlignment.spaceBetween,
                        children: [
                          const Text(
                            "Total",
                            style: TextStyle(
                              fontWeight: FontWeight.bold,
                            ),
                          ),
                          Text(
                            "Rp ${numberFormat.format(value.getTotalAmount())}",
                            style: const TextStyle(
                              fontWeight: FontWeight.bold,
                            ),
                          )
                        ],
                      ),
                    ),
                    const SizedBox(
                      height: 5.0,
                    ),
                    GestureDetector(
                      child: Container(
                        height: 0.1 * height,
                        color: Colors.blue.shade300,
                        child: const Center(
                          child: Text(
                            "PESAN SEKARANG",
                            style: TextStyle(
                              fontWeight: FontWeight.bold,
                            ),
                          ),
                        ),
                      ),
                      onTap: () {},
                    )
                  ],
                ),
              )
            : const Center(
                child: Text("Belum ada produk apapun"),
              );
      })),
    );
  }
}
```

Seperti biasa, agar dinamis, maka kita ambil `height` dengan menggunakan `MediaQuery`. Tujuannya adalah agar responsif.

Perhatikan, di sini kita memiliki `numberFormat`, tujuannya adalah untuk memformat uang sesuai dengan aturan Indonesia. Itulah kenapa kita menggunakan `id_ID`.

Kemudian, kita menggunakan `ListView` untuk menampilkan seluruh produk yang ada pada cart.

Terakhir, kita hanya perlu menampilkan total dari seluruh produk yang ada di cart kita. Caranya sangat mudah, kita hanya perlu memanggil fungsi yang sudah kita buat sebelumnya, yaitu `getTotalAmount`.

Selain itu, kita juga membuat sebuah tombol dengan label "PESAN SEKARANG". Supaya bisa diklik, maka kita gunakan `GestureDetector`.

# Mengubah Product Screen

---

Di sini, kita akan menambahkan `Navigator` agar ketika icon cart diklik, maka kita bisa pindah ke halaman cart. Sesuaikan source code-nya seperti di bawah ini:

```dart
import 'package:badges/badges.dart';
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';

import '../screens/cart_screen.dart';
import '../providers/cart_provider.dart';
import '../widgets/product_grid_widget.dart';

enum SelectedProductFilter {
  all,
  favoriteOnly,
}

class ProductScreen extends StatefulWidget {
  const ProductScreen({Key? key}) : super(key: key);

  @override
  State<ProductScreen> createState() => _ProductScreenState();
}

class _ProductScreenState extends State<ProductScreen> {
  bool selectedProductFilter = false;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        centerTitle: true,
        title: const Text("Shop App"),
        actions: [
          PopupMenuButton(
            onSelected: (SelectedProductFilter filter) {
              setState(() {
                if (filter == SelectedProductFilter.all) {
                  selectedProductFilter = false;
                } else {
                  selectedProductFilter = true;
                }
              });
            },
            icon: const Icon(Icons.more_horiz),
            itemBuilder: ((context) {
              return [
                const PopupMenuItem(
                  value: SelectedProductFilter.all,
                  child: Text("All"),
                ),
                const PopupMenuItem(
                  value: SelectedProductFilter.favoriteOnly,
                  child: Text("Favorite Only"),
                ),
              ];
            }),
          ),
          Consumer<CartProvider>(
            builder: ((context, value, child) {
              final cartLength =
                  context.read<CartProvider>().getCartLength().toString();

              return GestureDetector(
                onTap: () {
                  Navigator.pushNamed(context, CartScreen.routeName);
                },
                child: Badge(
                  badgeContent: Text(
                    cartLength,
                    style: const TextStyle(
                      color: Colors.black87,
                    ),
                  ),
                  position: BadgePosition.topEnd(top: 5, end: 15),
                  badgeColor: Colors.amber.shade200,
                  child: const Icon(Icons.shopping_cart),
                ),
              );
            }),
          )
        ],
      ),
      body: ProductGridWidget(
        selectedProductFilter: selectedProductFilter,
      ),
    );
  }
}
```

Perhatikan, di sini kita menggunakan `Navigator.pushNamed`, sesuai dengan apa yag sudah kita definisikan sebelumnya.

# Mengubah Product Widget

---

Di sini, kita akan memberikan feedback kepada user ketika mereka mengklik icon favorit atau icon cart dengan menggunakan Scaffold Messenger. Sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';

import '../models/product_model.dart';
import '../providers/cart_provider.dart';
import '../providers/product_provider.dart';
import '../screens/product_detail_screen.dart';

class ProductWidget extends StatelessWidget {
  final ProductModel product;

  const ProductWidget({
    Key? key,
    required this.product,
  }) : super(key: key);

  @override
  Widget build(BuildContext context) {
    final cartProvider = Provider.of<CartProvider>(context, listen: false);

    return ClipRRect(
      borderRadius: BorderRadius.circular(10.0),
      child: GridTile(
        footer: GridTileBar(
          backgroundColor: Colors.black54,
          leading: Consumer<ProductProvider>(builder: (context, value, child) {
            return IconButton(
              onPressed: () {
                context.read<ProductProvider>().toggleFavorite(product.id);

                if (value.getProductStatus(product.id)) {
                  ScaffoldMessenger.of(context).showSnackBar(
                    const SnackBar(
                      content: Text("Produk ditambahkan ke favorit"),
                    ),
                  );
                } else {
                  ScaffoldMessenger.of(context).showSnackBar(
                    const SnackBar(
                      content: Text("Produk dihapus dari favorit"),
                    ),
                  );
                }
              },
              icon: Icon(
                product.isFavorite ? Icons.favorite : Icons.favorite_border,
                color: Colors.blue.shade300,
              ),
            );
          }),
          title: Text(product.name),
          trailing: IconButton(
            onPressed: () {
              cartProvider.addProductToCart(product);

              ScaffoldMessenger.of(context).showSnackBar(
                const SnackBar(
                  content: Text("Produk ditambahkan ke cart"),
                ),
              );
            },
            icon: Icon(
              Icons.shopping_cart,
              color: Colors.blue.shade300,
            ),
          ),
        ),
        child: GestureDetector(
          child: Image.network(product.imageUrl),
          onTap: () {
            Navigator.pushNamed(
              context,
              ProductDetailScreen.routeName,
              arguments: product.id,
            );
          },
        ),
      ),
    );
  }
}
```

Perhatikan, ketika kita mengklik icon favorit atau cart, maka akan tampil sebuah pesan dari `ScaffoldMessenger`.

Berhubung icon favorit adalah toggle, maka kita perlu mengecek statusnya terlebih dahulu. Apakah nilai `isFavorite` nya itu `true` atau `false`. Bisa dilihat, kita memiliki pesan "ditambahkan" dan "dihapus".

Silakan ikuti langkah selanjutnya untuk mengubah `product_provider.dart`.

# Mengubah Product Provider

---

Di sini, kita akan menambahkan sebuah fungsi yang berguna untuk mengembalikan status favorit sebuah produk. Sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:flutter/material.dart';

import '../models/product_model.dart';

class ProductProvider with ChangeNotifier {
  // ignore: prefer_final_fields
  List<ProductModel> _products = [
    ProductModel(
      id: "id-1",
      name: "Macbook Air",
      description: "Macbook Air",
      price: 10000000,

          "https://images.pexels.com/photos/5202951/pexels-photo-5202951.jpeg?auto=compress&cs=tinysrgb&w=1260&h=750&dpr=1",
    ),
    ProductModel(
      id: "id-2",
      name: "Kamera Hitam dan Abu-Abu",
      description: "Kamera Hitam dan Abu-Abu",
      price: 7000000,
      imageUrl:
          "https://images.pexels.com/photos/1203803/pexels-photo-1203803.jpeg?auto=compress&cs=tinysrgb&w=1260&h=750&dpr=1",
    ),
    ProductModel(
      id: "id-3",
      name: "Flashdisk",
      description: "Flashdisk",
      price: 100000,
      imageUrl:
          "https://images.pexels.com/photos/10336136/pexels-photo-10336136.jpeg?auto=compress&cs=tinysrgb&w=1260&h=750&dpr=1",
    ),
    ProductModel(
      id: "id-4",
      name: "Keyboard",
      description: "Keyboard",
      price: 250000,
      imageUrl:
          "https://images.pexels.com/photos/585752/pexels-photo-585752.jpeg?auto=compress&cs=tinysrgb&w=1260&h=750&dpr=1",
    ),
  ];

  List<ProductModel> getProducts() {
    return _products;
  }

  List<ProductModel> getFavoriteOnlyProducts() {
    return _products.where((item) => item.isFavorite).toList();
  }

  void addProduct(ProductModel product) {
    _products.add(product);
    notifyListeners();
  }

  ProductModel findById(String id) {
    return _products.firstWhere((product) => product.id == id);
  }

  void toggleFavorite(String id) {
    ProductModel selectedProduct = findById(id);

    selectedProduct.isFavorite = !selectedProduct.isFavorite;

    notifyListeners();
  }

  bool getProductStatus(String id) {
    ProductModel currentProduct = findById(id);

    return currentProduct.isFavorite;
  }
}
```

Perhatikan, pada fungsi `getProductStatus`, kita mengembalikan nilai boolean dari variable `isFavorite`.
