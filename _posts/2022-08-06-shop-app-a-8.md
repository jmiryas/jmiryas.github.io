# Mengubah Cart Provider

---

```dart
import 'package:flutter/material.dart';

import '../models/cart_model.dart';
import '../models/product_model.dart';

class CartProvider extends ChangeNotifier {
  Map<String, CartModel> _carts = {};

  // Mendapatkan produk di cart

  Map<String, CartModel> getCarts() {
    return _carts;
  }

  // Mendapatkan banyak dari produk di cart

  int getCartLength() {
    return _carts.length;
  }

  // Mendapatkan total akhir dari produk di cart

  double getTotalAmount() {
    double total = 0.0;

    _carts.forEach((key, value) {
      total += value.quantity * value.price;
    });

    return total;
  }

  // Menambahkan produk ke cart

  void addProductToCart(ProductModel product) {
    if (_carts.containsKey(product.id)) {
      _carts.update(
        product.id,
        (value) => CartModel(
          id: value.id,
          name: value.name,
          price: value.price,
          quantity: value.quantity + 1,
        ),
      );
    } else {
      _carts.putIfAbsent(
        product.id,
        () => CartModel(
          id: DateTime.now().toString(),
          name: product.name,
          price: product.price,
          quantity: 1,
        ),
      );
    }

    notifyListeners();
  }

  // Remove produk dari cart berdasarkan id

  void removeById(String id) {
    _carts.removeWhere((key, value) => value.id == id);

    notifyListeners();
  }

  void clear() {
    _carts.clear();

    notifyListeners();
  }
}
```

Di sini, kita menambahkan fungsi `clear` untuk menghapus seluruh produk yang ada pada carts.

import '../models/cart_model.dart';

    required this.id,
    required this.amount,

# Membuat Order Provider

---

```dart
import 'package:flutter/material.dart';

import '../models/cart_model.dart';
import '../models/order_model.dart';

class OrderProvider with ChangeNotifier {
  List<OrderModel> _orders = [];

  List<OrderModel> getOrders() {
    return _orders;
  }

  void addOrder(double amount, List<CartModel> products) {
    _orders.add(OrderModel(
      id: DateTime.now().toString(),
      amount: amount,
      products: products,
      dateTime: DateTime.now(),
    ));
  }

  int getOrderLength() {
    return _orders.length;
  }
}
```

Fungsi `addOrder` berguna untuk menambahkan sebuah order ke dalam list orders. Sedangkan fungsi `getOrderLength` berguna untuk mendapatkan panjang dari list orders.

# Mengubah Main.dart

---

```dart
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';

import '../screens/cart_screen.dart';
import '../screens/order_screen.dart';
import '../screens/product_screen.dart';
import '../providers/cart_provider.dart';
import '../providers/order_provider.dart';
import '../providers/product_provider.dart';
import '../screens/product_detail_screen.dart';

void main() {
  runApp(
    MultiProvider(
      providers: [
        ChangeNotifierProvider(
          create: ((context) => ProductProvider()),
        ),
        ChangeNotifierProvider(
          create: ((context) => CartProvider()),
        ),
        ChangeNotifierProvider(create: (context) => OrderProvider())
      ],
      child: const MyApp(),
    ),
  );
}

class MyApp extends StatelessWidget {
  const MyApp({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: "Shop App",
      home: const ProductScreen(),
      initialRoute: ProductScreen.routeName,
      routes: {
        ProductDetailScreen.routeName: (context) => const ProductDetailScreen(),
        CartScreen.routeName: (context) => const CartScreen(),
        OrderScreen.routeName: (context) => const OrderScreen(),
      },
    );
  }
}
```

Di sini, kita menambahkan provider yang baru saja kita buat, yaitu `OrderProvider`.

# Membuat Order Screen

---

Pada folder `screens`, buat sebuah file baru dengan nama `order_screen.dart`. Sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';

import '../widgets/app_drawer.dart';
import '../models/order_model.dart';
import '../providers/order_provider.dart';
import '../widgets/order_item_widget.dart';

class OrderScreen extends StatelessWidget {
  static String routeName = "/order";

  const OrderScreen({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        centerTitle: true,
        title: const Text("Order"),
      ),
      body: Consumer<OrderProvider>(
        builder: ((context, value, child) {
          return value.getOrderLength() > 0
              ? Padding(
                  padding: const EdgeInsets.all(20.0),
                  child: ListView.builder(
                    itemBuilder: (context, index) {
                      OrderModel order = value.getOrders()[index];

                      return OrderItemWidget(order: order);
                    },
                    itemCount: value.getOrderLength(),
                  ),
                )
              : const Center(
                  child: Text("Belum ada order"),
                );
        }),
      ),
      drawer: const AppDrawer(),
    );
  }
}
```

Pertama, kita buat `routeName` pada `order_screen.dart`. Kemudian, kita gunakan `Consumer` untuk menampilkan seluruh produk yang ada pada orders.

Produk ini kita tampilkan dengan menggunakan `ListView`. Lalu, produk tersebut kita buat pada sebuah widget bernama `OrderItemWidget`.

Selain itu, terdapat juga sebuah drawer. Drawer ini berguna untuk menampilkan menu-menu yang kita miliki, yaitu Shop dan Order.

Tentu saja widget-widget ini belum ada, silakan ikuti langkah selanjutnya.

# Membuat Order Item Widget 

---

Pada folder `widgets`, buat sebuah file baru dengan nama `order_item_widget.dart`. Order item widget ini berguna untuk menampilkan produk yang ada pada halaman order. Sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:intl/intl.dart';
import 'package:flutter/material.dart';

import '../models/order_model.dart';

class OrderItemWidget extends StatelessWidget {
  final OrderModel order;

  const OrderItemWidget({
    Key? key,
    required this.order,
  }) : super(key: key);

  @override
  Widget build(BuildContext context) {
    final numberFormat = NumberFormat("#,##0.00", "id_ID");

    return Card(
      child: ExpansionTile(
        title: Text("Rp ${numberFormat.format(order.amount)}"),
        subtitle: Text(DateFormat("EEEE, MMMM d, yyyy").format(order.dateTime)),
        children: order.products.map((item) {
          return ListTile(
            title: Text(item.name),
            subtitle: Text("Rp ${numberFormat.format(item.price)}"),
            trailing: Text("${item.quantity}"),
          );
        }).toList(),
      ),
    );
  }
}
```

Di sini, kita menggunakan `ExpansionTile`. Tujuannya adalah agar bisa di-expand. Ketika di-expand, maka kita tampilkan produk-produk yang sudah diorder.

# Membuat Drawer Widget

---

Pada folder `widgets`, buat sebuah file baru dengan `app_drawer.dart`. Sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:flutter/material.dart';

import '../screens/order_screen.dart';
import '../screens/product_screen.dart';

class AppDrawer extends StatelessWidget {
  const AppDrawer({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Drawer(
      child: ListView(
        children: [
          ListTile(
            leading: const Icon(Icons.shopping_bag),
            title: const Text("Shop"),
            onTap: () {
              Navigator.pushNamed(context, ProductScreen.routeName);
            },
          ),
          ListTile(
            leading: const Icon(Icons.shop),
            title: const Text("Order"),
            onTap: () {
              Navigator.pushNamed(context, OrderScreen.routeName);
            },
          ),
        ],
      ),
    );
  }
}
```

Di sini, kita memiliki dua buah `ListTile`, itu artinya kita memiliki dua buah menu. Pertama untuk Shop, kedua untuk Order.

# Mengubah Cart Screen

---

Di sini, kita memanggil fungsi yang sudah kita definisikan sebelumnya pada `order_provider.dart`. Sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:intl/intl.dart';
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';
import 'package:shop_app/providers/order_provider.dart';
import 'package:shop_app/screens/order_screen.dart';

import '../providers/cart_provider.dart';
import '../widgets/cart_item_widget.dart';

class CartScreen extends StatelessWidget {
  static String routeName = "/cart";

  const CartScreen({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    double height = MediaQuery.of(context).size.height - 40.0 - 55.0;

    final numberFormat = NumberFormat("#,##0.00", "id_ID");

    return Scaffold(
      appBar: AppBar(
        toolbarHeight: 55.0,
        centerTitle: true,
        title: const Text("Cart"),
      ),
      body: Consumer<CartProvider>(
        builder: ((context, value, child) {
          return value.getCartLength() > 0
              ? Padding(
                  padding: const EdgeInsets.all(20.0),
                  child: Column(
                    children: [
                      // Perulangan untuk menampilkan
                      // seluruh produk pada cart

                      Expanded(
                        child: ListView(
                          children: value.getCarts().entries.map((item) {
                            String price =
                                numberFormat.format(item.value.price);

                            return Column(
                              children: [
                                CartItemWidget(
                                  id: item.value.id,
                                  name: item.value.name,
                                  price: price,
                                  quantity: item.value.quantity,
                                ),
                              ],
                            );
                          }).toList(),
                        ),
                      ),
                      const SizedBox(
                        height: 10.0,
                      ),
                      Container(
                        height: 0.1 * height,
                        color: Colors.grey.shade300,
                        padding: const EdgeInsets.symmetric(horizontal: 10.0),
                        child: Row(
                          mainAxisAlignment: MainAxisAlignment.spaceBetween,
                          children: [
                            const Text(
                              "Total",
                              style: TextStyle(
                                fontWeight: FontWeight.bold,
                              ),
                            ),
                            Text(
                              "Rp ${numberFormat.format(value.getTotalAmount())}",
                              style: const TextStyle(
                                fontWeight: FontWeight.bold,
                              ),
                            )
                          ],
                        ),
                      ),
                      const SizedBox(
                        height: 5.0,
                      ),
                      GestureDetector(
                        child: Container(
                          height: 0.1 * height,
                          color: Colors.blue.shade300,
                          child: const Center(
                            child: Text(
                              "PESAN SEKARANG",
                              style: TextStyle(
                                fontWeight: FontWeight.bold,
                              ),
                            ),
                          ),
                        ),
                        onTap: () {
                          Provider.of<OrderProvider>(context, listen: false)
                              .addOrder(
                            value.getTotalAmount(),
                            value.getCarts().values.toList(),
                          );

                          context.read<CartProvider>().clear();

                          Navigator.pushNamed(context, OrderScreen.routeName);
                        },
                      )
                    ],
                  ),
                )
              : const Center(
                  child: Text("Belum ada produk apapun"),
                );
        }),
      ),
    );
  }
}
```
Pada `onTap`, kita panggil fungsi `addOrder`. Kemudian, kita panggil fungsi `clear` pada `cart_provider.dart`. Terkahir, kita pindahkan halaman ke halaman order.
