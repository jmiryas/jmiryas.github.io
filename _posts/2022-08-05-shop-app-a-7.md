# Mengubah Cart Provider

---

Di sini, kita akan menambahkan sebuah fungsi untuk menghapus produk yang ada di cart berdasarkan id. Sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:flutter/material.dart';

import '../models/cart_model.dart';
import '../models/product_model.dart';

class CartProvider extends ChangeNotifier {
  Map<String, CartModel> _carts = {};

  // Mendapatkan produk di cart

  Map<String, CartModel> getCarts() {
    return _carts;
  }

  // Mendapatkan banyak dari produk di cart

  int getCartLength() {
    return _carts.length;
  }

  // Mendapatkan total akhir dari produk di cart

  double getTotalAmount() {
    double total = 0.0;

    _carts.forEach((key, value) {
      total += value.quantity * value.price;
    });

    return total;
  }

  // Menambahkan produk ke cart

  void addProductToCart(ProductModel product) {
    if (_carts.containsKey(product.id)) {
      _carts.update(
        product.id,
        (value) => CartModel(
          id: value.id,
          name: value.name,
          price: value.price,
          quantity: value.quantity + 1,
        ),
      );
    } else {
      _carts.putIfAbsent(
        product.id,
        () => CartModel(
          id: DateTime.now().toString(),
          name: product.name,
          price: product.price,
          quantity: 1,
        ),
      );
    }

    notifyListeners();
  }

  // Remove produk dari cart berdasarkan id

  void removeById(String id) {
    _carts.removeWhere((key, value) => value.id == id);

    notifyListeners();
  }
}
```

Perhatikan pada fungsi `removeById`. Di sini, kita menghapus sebuah produk yang ada di cart berdasarkan id produk yang diberikan.

# Mengubah Cart Screen

---

Di sini, kita akan memindahkan block of code pada sebuah widget yang bernama `CartItemWidget`. Sesuaikan source code-nya seperti di bawah ini:

```dart
import 'package:intl/intl.dart';
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';

import '../providers/cart_provider.dart';
import '../widgets/cart_item_widget.dart';

class CartScreen extends StatelessWidget {
  static String routeName = "/cart";

  const CartScreen({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    double height = MediaQuery.of(context).size.height - 40.0 - 55.0;

    final numberFormat = NumberFormat("#,##0.00", "id_ID");

    return Scaffold(
      appBar: AppBar(
        toolbarHeight: 55.0,
        centerTitle: true,
        title: const Text("Cart"),
      ),
      body: Consumer<CartProvider>(
        builder: ((context, value, child) {
          return value.getCartLength() > 0
              ? Padding(
                  padding: const EdgeInsets.all(20.0),
                  child: Column(
                    children: [
                      // Perulangan untuk menampilkan
                      // seluruh produk pada cart

                      SizedBox(
                        height: 0.5 * height,
                        child: ListView(
                          children: value.getCarts().entries.map((item) {
                            String price =
                                numberFormat.format(item.value.price);

                            return Column(
                              children: [
                                CartItemWidget(
                                  id: item.value.id,
                                  name: item.value.name,
                                  price: price,
                                  quantity: item.value.quantity,
                                ),
                                const Divider(
                                  color: Colors.black26,
                                )
                              ],
                            );
                          }).toList(),
                        ),
                      ),
                      const SizedBox(
                        height: 20.0,
                      ),
                      Container(
                        height: 0.1 * height,
                        color: Colors.grey.shade300,
                        padding: const EdgeInsets.symmetric(horizontal: 10.0),
                        child: Row(
                          mainAxisAlignment: MainAxisAlignment.spaceBetween,
                          children: [
                            const Text(
                              "Total",
                              style: TextStyle(
                                fontWeight: FontWeight.bold,
                              ),
                            ),
                            Text(
                              "Rp ${numberFormat.format(value.getTotalAmount())}",
                              style: const TextStyle(
                                fontWeight: FontWeight.bold,
                              ),
                            )
                          ],
                        ),
                      ),
                      const SizedBox(
                        height: 5.0,
                      ),
                      GestureDetector(
                        child: Container(
                          height: 0.1 * height,
                          color: Colors.blue.shade300,
                          child: const Center(
                            child: Text(
                              "PESAN SEKARANG",
                              style: TextStyle(
                                fontWeight: FontWeight.bold,
                              ),
                            ),
                          ),
                        ),
                        onTap: () {},
                      )
                    ],
                  ),
                )
              : const Center(
                  child: Text("Belum ada produk apapun"),
                );
        }),
      ),
    );
  }
}
```

Tentu saja widget ini belum ada, maka kita harus membuatnya terlebih dahulu. Silakan ikuti langkah selanjutnya.

# Membuat Cart Item Widget

---

Buka folder `widgets`, lalu buat sebuah file baru dengan nama `cart_item_widget.dart`. Sesuaikan source code-nya menjadi seperti ini:

```dart
import "package:flutter/material.dart";
import 'package:provider/provider.dart';

import '../providers/cart_provider.dart';

class CartItemWidget extends StatelessWidget {
  final String id;
  final String name;
  final String price;
  final int quantity;

  const CartItemWidget({
    Key? key,
    required this.id,
    required this.name,
    required this.price,
    required this.quantity,
  }) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Dismissible(
      key: ValueKey(id),
      onDismissed: (DismissDirection direction) {
        Provider.of<CartProvider>(context, listen: false).removeById(id);

        ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(
            content: Text("Produk dihapus dari cart"),
          ),
        );
      },
      background: Container(
        padding: const EdgeInsets.only(left: 10.0),
        color: Colors.red.shade300,
        alignment: Alignment.centerLeft,
        child: const Icon(
          Icons.delete,
          color: Colors.white,
        ),
      ),
      direction: DismissDirection.startToEnd,
      child: ListTile(
        title: Text(
          name,
          style: const TextStyle(
            fontWeight: FontWeight.bold,
          ),
        ),
        subtitle: Text("Rp $price"),
        trailing: Text("$quantity"),
      ),
    );
  }
}
```

Perhatikan, `ListTile` yang sebelumnya ada di `cart_screen.dart`, sekarang sudah kita pindahkan ke sini.

Kemudian, kita menggunakan `Dismissible` agar data bisa dihapus ketika melakukan swipe. Arah swipe kita atur dari kiri ke kanan dengan menggunakan `DismissDirection.startToEnd`.

Pada saat swipe terjadi, kita tampilkan background berwarna merah dengan icon delete agar user memiliki gambaran tentang apa yang sedang mereka lakukan.

Selanjutnya, pada `onDismissed`, kita panggil fungsi `removeById` yang sebelumnya sudah kita definisikan.

# Mengubah Product Detail Screen

---

Di sini, kita hanya perlu menambahkan kontennya saja. Sesuaikan source code-nya menjadi seperti ini:

```dart
import 'package:intl/intl.dart';
import 'package:flutter/material.dart';
import 'package:provider/provider.dart';

import '../providers/product_provider.dart';

class ProductDetailScreen extends StatelessWidget {
  static String routeName = "/product-detail";

  const ProductDetailScreen({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    final productId = ModalRoute.of(context)!.settings.arguments as String;

    final currentProduct =
        Provider.of<ProductProvider>(context).findById(productId);

    final numberFormat = NumberFormat("#,##0.00", "id_ID");

    double height = MediaQuery.of(context).size.height - 40.0 - 55.0;

    return Scaffold(
      appBar: AppBar(
        centerTitle: true,
        title: Text(currentProduct.name),
      ),
      body: Padding(
        padding: const EdgeInsets.all(20.0),
        child: Column(
          children: [
            SizedBox(
              width: double.infinity,
              height: 0.4 * height,
              child: ClipRRect(
                borderRadius: BorderRadius.circular(10.0),
                child: Image.network(
                  currentProduct.imageUrl,
                  fit: BoxFit.cover,
                ),
              ),
            ),
            const SizedBox(
              height: 10.0,
            ),
            SizedBox(
              width: double.infinity,
              child: Text(
                currentProduct.name,
                style: const TextStyle(
                  fontWeight: FontWeight.bold,
                  fontSize: 20.0,
                ),
                overflow: TextOverflow.ellipsis,
                maxLines: 2,
              ),
            ),
            SizedBox(
              width: double.infinity,
              child: Text(
                "Rp ${numberFormat.format(currentProduct.price)}",
                style: const TextStyle(
                  fontSize: 15.0,
                ),
              ),
            )
          ],
        ),
      ),
    );
  }
}
```

Pada source code di atas, kita menampilkan gambar, nama dan harga dari produk.
